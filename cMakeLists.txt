cmake_minimum_required(VERSION 3.10) # 指定CMake的最低版本要求
project(2048) # 指定项目的名称

set(CMAKE_C_COMPILER "/e/MSYS2/mingw64/bin/gcc")
set(CMAKE_CXX_COMPILER "/e/MSYS2/mingw64/bin/g++")

set(CMAKE_CXX_STANDARD 14) # 指定C++的版本

# 添加头文件搜索路径
include_directories(freeMap Generated Hide Initializing move over Paint run test)

# 创建一个空的SOURCES变量
set(SOURCES "")

# 添加所有的.cpp文件到SOURCES变量
file(GLOB_RECURSE TEMP_SOURCES "Initializing/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "Hide/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "Paint/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "over/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "Generated/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "move/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "test/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "freeMap/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

file(GLOB_RECURSE TEMP_SOURCES "run/*.cpp")
list(APPEND SOURCES ${TEMP_SOURCES})

list(APPEND SOURCES "main.cpp")

# 添加所有的源文件到一个名为2048的可执行文件
add_executable(2048 ${SOURCES})
set_target_properties(2048 PROPERTIES RUNTIME_OUTPUT_DIRECTORY /g/CODE/2048)
